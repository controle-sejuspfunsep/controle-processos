Contas orçamentárias create table if not exists fin_contas ( id uuid default gen_random_uuid() primary key, numero text not null, ano_repasse integer not null, nd text not null, -- '3' ou '4' eixo text not null, -- EVC, VPSP, FISP, etc. acao text not null, saldo_inicial numeric(14,2) default 0, saldo_conta numeric(14,2) default 0, created_at timestamptz default now() ); alter table fin_contas enable row level security; drop policy if exists "fin_contas_policy" on fin_contas; create policy "fin_contas_policy" on fin_contas for all using (auth.role() = 'authenticated') with check (auth.role() = 'authenticated'); -- 2. Empenhos create table if not exists fin_empenhos ( id uuid default gen_random_uuid() primary key, numero text not null, data_empenho date, conta_id uuid references fin_contas(id), processo_siga text, valor numeric(14,2) not null, situacao text default 'ativo', obs text, created_at timestamptz default now() ); alter table fin_empenhos enable row level security; drop policy if exists "fin_empenhos_policy" on fin_empenhos; create policy "fin_empenhos_policy" on fin_empenhos for all using (auth.role() = 'authenticated') with check (auth.role() = 'authenticated'); -- 3. Realtime do $$ begin begin alter publication supabase_realtime add table fin_contas; exception when others then null; end; begin alter publication supabase_realtime add table fin_empenhos; exception when others then null; end; end $$; -- Confirmação select table_name, 'OK' as status from information_schema.tables where table_schema = 'public' and table_name in ('fin_contas','fin_empenhos') order by table_name;
